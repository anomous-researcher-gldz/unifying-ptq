# CompSRT/basicsr/quant/rtn_baseline.py
"""Weak SR PTQ baseline: per-tensor RTN quantization of all linear/conv weights
in a SwinIR model, with optional DBAF folding pre-quantization.
"""
from __future__ import annotations
import torch
import torch.nn as nn

def _dbaf_fold(w, alpha=0.95, T_sigma=3.0):
    sigma = w.std()
    T = T_sigma * sigma
    sgn = torch.sign(w); mask = w.abs() > T
    out = w.clone()
    out[mask] = sgn[mask] * T + alpha * (w[mask] - sgn[mask] * T)
    return out, T, alpha

def _dbaf_unfold(w_q, T, alpha):
    sgn = torch.sign(w_q); mask = w_q.abs() > T
    out = w_q.clone()
    out[mask] = sgn[mask] * T + (1.0 / alpha) * (w_q[mask] - sgn[mask] * T)
    return out

def _quant(w, bits):
    qmax = 2 ** (bits - 1) - 1
    s = w.abs().max() / qmax
    if s == 0: return w
    q = torch.round(w / s).clamp(-qmax, qmax)
    return (q * s).to(w.dtype)

def quantize_swinir_rtn(model: nn.Module, bits: int = 4, use_dbaf: bool = False, alpha: float = 0.95) -> nn.Module:
    """In-place RTN quantization with optional DBAF for SwinIR."""
    for name, mod in model.named_modules():
        if not isinstance(mod, (nn.Linear, nn.Conv2d)):
            continue
        # Skip the final upsampler/output conv to preserve fidelity
        if "conv_last" in name or "upsample" in name:
            continue
        w = mod.weight.data
        if use_dbaf:
            w_fold, T, a = _dbaf_fold(w, alpha=alpha)
            w_q = _quant(w_fold, bits)
            w_out = _dbaf_unfold(w_q, T, a)
        else:
            w_out = _quant(w, bits)
        mod.weight.data = w_out.to(mod.weight.dtype)
    return model
```

# CompSRT/scripts/eval_swinir_set5.py
"""Evaluate a SwinIR model (FP, RTN, RTN+DBAF, torchao) on Set5 for PSNR/SSIM."""
import argparse, json, pathlib, sys
import torch
from torch.utils.data import DataLoader
sys.path.insert(0, "/home/ubuntu/unifying-ptq/CompSRT")

def load_set5(scale, root="/home/ubuntu/unifying-ptq/CompSRT/datasets/Set5"):
    from basicsr.data.paired_image_dataset import PairedImageDataset
    opt = {
        "name": "Set5", "type": "PairedImageDataset",
        "dataroot_gt": f"{root}/GTmod{scale}", "dataroot_lq": f"{root}/LRbicx{scale}",
        "filename_tmpl": "{}", "io_backend": {"type": "disk"},
        "scale": scale, "phase": "test",
    }
    return PairedImageDataset(opt)

def eval_psnr_ssim(model, loader):
    from basicsr.metrics import calculate_psnr, calculate_ssim
    psnrs, ssims = [], []
    for batch in loader:
        lq = batch["lq"].cuda().half()
        gt = batch["gt"].cuda().half()
        with torch.no_grad():
            sr = model(lq)
        sr_np = (sr.clamp(0,1)[0].permute(1,2,0).cpu().numpy() * 255).astype("uint8")
        gt_np = (gt.clamp(0,1)[0].permute(1,2,0).cpu().numpy() * 255).astype("uint8")
        psnrs.append(calculate_psnr(sr_np, gt_np, crop_border=0, test_y_channel=True))
        ssims.append(calculate_ssim(sr_np, gt_np, crop_border=0, test_y_channel=True))
    return sum(psnrs)/len(psnrs), sum(ssims)/len(ssims)

def main():
    p = argparse.ArgumentParser()
    p.add_argument("--checkpoint", required=True)
    p.add_argument("--method", choices=["fp", "rtn", "rtn-dbaf", "torchao-w4a4", "torchao-w4a16"], required=True)
    p.add_argument("--scale", type=int, default=2)
    p.add_argument("--out", required=True)
    args = p.parse_args()

    from basicsr.archs.swinir_arch import SwinIR
    m = SwinIR(upscale=args.scale, in_chans=3, img_size=48, window_size=8, img_range=1.,
               depths=[6,6,6,6,6,6], embed_dim=60, num_heads=[6,6,6,6,6,6], mlp_ratio=2,
               upsampler='pixelshuffledirect', resi_connection='1conv')
    state = torch.load(args.checkpoint, map_location="cpu")
    m.load_state_dict(state["params"] if "params" in state else state, strict=False)
    m = m.cuda().half(); m.eval()

    if args.method == "fp":
        pass
    elif args.method == "rtn":
        from basicsr.quant.rtn_baseline import quantize_swinir_rtn
        m = quantize_swinir_rtn(m, bits=4, use_dbaf=False)
    elif args.method == "rtn-dbaf":
        from basicsr.quant.rtn_baseline import quantize_swinir_rtn
        m = quantize_swinir_rtn(m, bits=4, use_dbaf=True)
    elif args.method.startswith("torchao"):
        from basicsr.torchao_deploy import apply_torchao_swinir
        config_name = args.method.split("-")[-1]  # w4a4 or w4a16
        m = apply_torchao_swinir(m, config_name=config_name)

    loader = DataLoader(load_set5(args.scale), batch_size=1, shuffle=False, num_workers=2)
    psnr, ssim = eval_psnr_ssim(m, loader)
    out = {"model":"SwinIR","scale":args.scale,"method":args.method,
           "psnr":psnr,"ssim":ssim}
    pathlib.Path(args.out).parent.mkdir(parents=True, exist_ok=True)
    pathlib.Path(args.out).write_text(json.dumps(out, indent=2))
    print(out)

if __name__ == "__main__":
    main()
```
